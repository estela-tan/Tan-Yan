def plot_data(df, x_col, y_col, title, ax):
    ax.plot(df[x_col], df[y_col], linewidth=1)
    
    ax.set_title(title)
    ax.set_xlabel(x_col)
    ax.set_ylabel(y_col.replace('_', ' ').capitalize())
    
    ax.grid(True)
 
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(14, 6))
